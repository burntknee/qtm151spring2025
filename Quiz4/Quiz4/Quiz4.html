currencies["bool_cond"]  = (currencies["growth_euro"] > 2) | (currencies["growth_euro"] < -2)
plt.plot("datetime", "growth_euro", data = currencies)
plt.xlabel("Date")
plt.ylabel("Euro to USD Growth Rate")
plt.title("Euro to USD Growth Rate > 2 or < -2")

vec_y = currencies["growth_euro"]
plt.fill_between(x= currencies["datetime"],
                 y1 = vec_y.min(),
                 y2 = vec_y.max(),
                 where = currencies["bool_cond"],
                 alpha = 0.2,color = "red")

plt.show()
